173// Backend data stored in io.BackendPlatformUserData to allow support for multiple Dear ImGui contexts
174// It is STRONGLY preferred that you use docking branch with multi-viewports (== single Dear ImGui context + multiple windows) instead of multiple Dear ImGui contexts.
175// FIXME: multi-context support is not well tested and probably dysfunctional in this backend.
176// - Because glfwPollEvents() process all windows and some events may be called outside of it, you will need to register your own callbacks
177// (passing install_callbacks=false in ImGui_ImplGlfw_InitXXX functions), set the current dear imgui context and then call our callbacks.
178// - Otherwise we may need to store a GLFWWindow* -> ImGuiContext* map and handle this in the backend, adding a little bit of extra complexity to it.
179// FIXME: some shared resources (mouse cursor shape, gamepad) are mishandled when using multi-context.
180static ImGui_ImplGlfw_Data* ImGui_ImplGlfw_GetBackendData()
181{
182 return ImGui::GetCurrentContext() ? (ImGui_ImplGlfw_Data*)ImGui::GetIO().BackendPlatformUserData : nullptr;
183}

326// X11 does not include current pressed/released modifier key in 'mods' flags submitted by GLFW
327// See https://github.com/ocornut/imgui/issues/6034 and https://github.com/glfw/glfw/issues/1630
328static void ImGui_ImplGlfw_UpdateKeyModifiers(GLFWwindow* window)
329{
330 ImGuiIO& io = ImGui::GetIO();
331 io.AddKeyEvent(ImGuiMod_Ctrl, (glfwGetKey(window, GLFW_KEY_LEFT_CONTROL) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_CONTROL) == GLFW_PRESS));
332 io.AddKeyEvent(ImGuiMod_Shift, (glfwGetKey(window, GLFW_KEY_LEFT_SHIFT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SHIFT) == GLFW_PRESS));
333 io.AddKeyEvent(ImGuiMod_Alt, (glfwGetKey(window, GLFW_KEY_LEFT_ALT) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_ALT) == GLFW_PRESS));
334 io.AddKeyEvent(ImGuiMod_Super, (glfwGetKey(window, GLFW_KEY_LEFT_SUPER) == GLFW_PRESS) || (glfwGetKey(window, GLFW_KEY_RIGHT_SUPER) == GLFW_PRESS));
335}

343void ImGui_ImplGlfw_MouseButtonCallback(GLFWwindow* window, int button, int action, int mods)
344{
345 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
346 if (bd->PrevUserCallbackMousebutton != nullptr && ImGui_ImplGlfw_ShouldChainCallback(window))
347 bd->PrevUserCallbackMousebutton(window, button, action, mods);
348
349 ImGui_ImplGlfw_UpdateKeyModifiers(window);
350
351 ImGuiIO& io = ImGui::GetIO();
352 if (button >= 0 && button < ImGuiMouseButton_COUNT)
353 io.AddMouseButtonEvent(button, action == GLFW_PRESS);
354}

374 // GLFW 3.1+ attempts to "untranslate" keys, which goes the opposite of what every other framework does, making using lettered shortcuts difficult.
375 // (It had reasons to do so: namely GLFW is/was more likely to be used for WASD-type game controls rather than lettered shortcuts, but IHMO the 3.1 change could have been done differently)
376 // See https://github.com/glfw/glfw/issues/1502 for details.
377 // Adding a workaround to undo this (so our keys are translated->untranslated->translated, likely a lossy process).
378 // This won't cover edge cases but this is at least going to cover common cases.
379 if (key >= GLFW_KEY_KP_0 && key <= GLFW_KEY_KP_EQUAL)
380 return key;
381 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
382 const char* key_name = glfwGetKeyName(key, scancode);
383 glfwSetErrorCallback(prev_error_callback);
384#if GLFW_HAS_GETERROR && !defined(__EMSCRIPTEN__) // Eat errors (see #5908)
385 (void)glfwGetError(nullptr);
386#endif
387 if (key_name && key_name[0] != 0 && key_name[1] == 0)
388 {
389 const char char_names[] = "`-=[]\\,;\'./";
390 const int char_keys[] = { GLFW_KEY_GRAVE_ACCENT, GLFW_KEY_MINUS, GLFW_KEY_EQUAL, GLFW_KEY_LEFT_BRACKET, GLFW_KEY_RIGHT_BRACKET, GLFW_KEY_BACKSLASH, GLFW_KEY_COMMA, GLFW_KEY_SEMICOLON, GLFW_KEY_APOSTROPHE, GLFW_KEY_PERIOD, GLFW_KEY_SLASH, 0 };
391 IM_ASSERT(IM_ARRAYSIZE(char_names) == IM_ARRAYSIZE(char_keys));
392 if (key_name[0] >= '0' && key_name[0] <= '9') { key = GLFW_KEY_0 + (key_name[0] - '0'); }
393 else if (key_name[0] >= 'A' && key_name[0] <= 'Z') { key = GLFW_KEY_A + (key_name[0] - 'A'); }
394 else if (key_name[0] >= 'a' && key_name[0] <= 'z') { key = GLFW_KEY_A + (key_name[0] - 'a'); }
395 else if (const char* p = strchr(char_names, key_name[0])) { key = char_keys[p - char_names]; }
396 }
397 // if (action == GLFW_PRESS) printf("key %d scancode %d name '%s'\n", key, scancode, key_name);
398#else
399 IM_UNUSED(scancode);
400#endif
401 return key;
402}

566static bool ImGui_ImplGlfw_Init(GLFWwindow* window, bool install_callbacks, GlfwClientApi client_api)
567{
568 ImGuiIO& io = ImGui::GetIO();
569 IM_ASSERT(io.BackendPlatformUserData == nullptr && "Already initialized a platform backend!");
570 //printf("GLFW_VERSION: %d.%d.%d (%d)", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION, GLFW_VERSION_COMBINED);
571
572 // Setup backend capabilities flags
573 ImGui_ImplGlfw_Data* bd = IM_NEW(ImGui_ImplGlfw_Data)();
574 io.BackendPlatformUserData = (void*)bd;
575 io.BackendPlatformName = "imgui_impl_glfw";
576 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
577 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
578#ifndef __EMSCRIPTEN__
579 io.BackendFlags |= ImGuiBackendFlags_PlatformHasViewports; // We can create multi-viewports on the Platform side (optional)
580#endif
581#if GLFW_HAS_MOUSE_PASSTHROUGH || (GLFW_HAS_WINDOW_HOVERED && defined(_WIN32))
582 io.BackendFlags |= ImGuiBackendFlags_HasMouseHoveredViewport; // We can call io.AddMouseViewportEvent() with correct data (optional)
583#endif
584
585 bd->Window = window;
586 bd->Time = 0.0;
587 bd->WantUpdateMonitors = true;
588
589 io.SetClipboardTextFn = ImGui_ImplGlfw_SetClipboardText;
590 io.GetClipboardTextFn = ImGui_ImplGlfw_GetClipboardText;
591 io.ClipboardUserData = bd->Window;
592
593 // Create mouse cursors
594 // (By design, on X11 cursors are user configurable and some cursors may be missing. When a cursor doesn't exist,
595 // GLFW will emit an error which will often be printed by the app, so we temporarily disable error reporting.
596 // Missing cursors will return nullptr and our _UpdateMouseCursor() function will use the Arrow cursor instead.)
597 GLFWerrorfun prev_error_callback = glfwSetErrorCallback(nullptr);
598 bd->MouseCursors[ImGuiMouseCursor_Arrow] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
599 bd->MouseCursors[ImGuiMouseCursor_TextInput] = glfwCreateStandardCursor(GLFW_IBEAM_CURSOR);
600 bd->MouseCursors[ImGuiMouseCursor_ResizeNS] = glfwCreateStandardCursor(GLFW_VRESIZE_CURSOR);
601 bd->MouseCursors[ImGuiMouseCursor_ResizeEW] = glfwCreateStandardCursor(GLFW_HRESIZE_CURSOR);
602 bd->MouseCursors[ImGuiMouseCursor_Hand] = glfwCreateStandardCursor(GLFW_HAND_CURSOR);
603#if GLFW_HAS_NEW_CURSORS
604 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_RESIZE_ALL_CURSOR);
605 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_RESIZE_NESW_CURSOR);
606 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_RESIZE_NWSE_CURSOR);
607 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_NOT_ALLOWED_CURSOR);
608#else
609 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
610 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
611 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
612 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = glfwCreateStandardCursor(GLFW_ARROW_CURSOR);
613#endif
614 glfwSetErrorCallback(prev_error_callback);
615#if GLFW_HAS_GETERROR && !defined(__EMSCRIPTEN__) // Eat errors (see #5908)
616 (void)glfwGetError(nullptr);
617#endif
618
619 // Chain GLFW callbacks: our callbacks will call the user's previously installed callbacks, if any.
620 if (install_callbacks)
621 ImGui_ImplGlfw_InstallCallbacks(window);
622 // Register Emscripten Wheel callback to workaround issue in Emscripten GLFW Emulation (#6096)
623 // We intentionally do not check 'if (install_callbacks)' here, as some users may set it to false and call GLFW callback themselves.
624 // FIXME: May break chaining in case user registered their own Emscripten callback?
625#ifdef __EMSCRIPTEN__
626 emscripten_set_wheel_callback(EMSCRIPTEN_EVENT_TARGET_DOCUMENT, nullptr, false, ImGui_ImplEmscripten_WheelCallback);
627#endif
628
629 // Update monitors the first time (note: monitor callback are broken in GLFW 3.2 and earlier, see github.com/glfw/glfw/issues/784)
630 ImGui_ImplGlfw_UpdateMonitors();
631 glfwSetMonitorCallback(ImGui_ImplGlfw_MonitorCallback);
632
633 // Set platform dependent data in viewport
634 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
635 main_viewport->PlatformHandle = (void*)bd->Window;
636#ifdef _WIN32
637 main_viewport->PlatformHandleRaw = glfwGetWin32Window(bd->Window);
638#elif defined(__APPLE__)
639 main_viewport->PlatformHandleRaw = (void*)glfwGetCocoaWindow(bd->Window);
640#else
641 IM_UNUSED(main_viewport);
642#endif
643 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
644 ImGui_ImplGlfw_InitPlatformInterface();
645
646 // Windows: register a WndProc hook so we can intercept some messages.
647#ifdef _WIN32
648 bd->PrevWndProc = (WNDPROC)::GetWindowLongPtr((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC);
649 IM_ASSERT(bd->PrevWndProc != nullptr);
650 ::SetWindowLongPtr((HWND)main_viewport->PlatformHandleRaw, GWLP_WNDPROC, (LONG_PTR)ImGui_ImplGlfw_WndProc);
651#endif
652
653 bd->ClientApi = client_api;
654 return true;
655}

702static void ImGui_ImplGlfw_UpdateMouseData()
703{
704 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
705 ImGuiIO& io = ImGui::GetIO();
706 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
707
708
709 ImGuiID mouse_viewport_id = 0;
710 const ImVec2 mouse_pos_prev = io.MousePos;
711 for (int n = 0; n < platform_io.Viewports.Size; n++)
712 {
713 ImGuiViewport* viewport = platform_io.Viewports[n];
714 GLFWwindow* window = (GLFWwindow*)viewport->PlatformHandle;
715
716#ifdef __EMSCRIPTEN__
717 const bool is_window_focused = true;
718#else
719 const bool is_window_focused = glfwGetWindowAttrib(window, GLFW_FOCUSED) != 0;
720#endif
721 if (is_window_focused)
722 {
723 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
724 // When multi-viewports are enabled, all Dear ImGui positions are same as OS positions.
725 if (io.WantSetMousePos)
726 glfwSetCursorPos(window, (double)(mouse_pos_prev.x - viewport->Pos.x), (double)(mouse_pos_prev.y - viewport->Pos.y));
727
728 // (Optional) Fallback to provide mouse position when focused (ImGui_ImplGlfw_CursorPosCallback already provides this when hovered or captured)
729 if (bd->MouseWindow == nullptr)
730 {
731 double mouse_x, mouse_y;
732 glfwGetCursorPos(window, &mouse_x, &mouse_y);
733 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
734 {
735 // Single viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
736 // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
737 int window_x, window_y;
738 glfwGetWindowPos(window, &window_x, &window_y);
739 mouse_x += window_x;
740 mouse_y += window_y;
741 }
742 bd->LastValidMousePos = ImVec2((float)mouse_x, (float)mouse_y);
743 io.AddMousePosEvent((float)mouse_x, (float)mouse_y);
744 }
745 }
746
747 // (Optional) When using multiple viewports: call io.AddMouseViewportEvent() with the viewport the OS mouse cursor is hovering.
748 // If ImGuiBackendFlags_HasMouseHoveredViewport is not set by the backend, Dear imGui will ignore this field and infer the information using its flawed heuristic.
749 // - [X] GLFW >= 3.3 backend ON WINDOWS ONLY does correctly ignore viewports with the _NoInputs flag.
750 // - [!] GLFW <= 3.2 backend CANNOT correctly ignore viewports with the _NoInputs flag, and CANNOT reported Hovered Viewport because of mouse capture.
751 // Some backend are not able to handle that correctly. If a backend report an hovered viewport that has the _NoInputs flag (e.g. when dragging a window
752 // for docking, the viewport has the _NoInputs flag in order to allow us to find the viewport under), then Dear ImGui is forced to ignore the value reported
753 // by the backend, and use its flawed heuristic to guess the viewport behind.
754 // - [X] GLFW backend correctly reports this regardless of another viewport behind focused and dragged from (we need this to find a useful drag and drop target).
755 // FIXME: This is currently only correct on Win32. See what we do below with the WM_NCHITTEST, missing an equivalent for other systems.
756 // See https://github.com/glfw/glfw/issues/1236 if you want to help in making this a GLFW feature.
757#if GLFW_HAS_MOUSE_PASSTHROUGH || (GLFW_HAS_WINDOW_HOVERED && defined(_WIN32))
758 const bool window_no_input = (viewport->Flags & ImGuiViewportFlags_NoInputs) != 0;
759#if GLFW_HAS_MOUSE_PASSTHROUGH
760 glfwSetWindowAttrib(window, GLFW_MOUSE_PASSTHROUGH, window_no_input);
761#endif
762 if (glfwGetWindowAttrib(window, GLFW_HOVERED) && !window_no_input)
763 mouse_viewport_id = viewport->ID;
764#else
765 // We cannot use bd->MouseWindow maintained from CursorEnter/Leave callbacks, because it is locked to the window capturing mouse.
766#endif
767 }
768
769 if (io.BackendFlags & ImGuiBackendFlags_HasMouseHoveredViewport)
770 io.AddMouseViewportEvent(mouse_viewport_id);
771}

800// Update gamepad inputs
801static inline float Saturate(float v) { return v < 0.0f ? 0.0f : v > 1.0f ? 1.0f : v; }
802static void ImGui_ImplGlfw_UpdateGamepads()
803{
804 ImGuiIO& io = ImGui::GetIO();
805 if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0) // FIXME: Technically feeding gamepad shouldn't depend on this now that they are regular inputs.
806 return;
807
808 io.BackendFlags &= ~ImGuiBackendFlags_HasGamepad;
809#if GLFW_HAS_GAMEPAD_API && !defined(__EMSCRIPTEN__)
810 GLFWgamepadstate gamepad;
811 if (!glfwGetGamepadState(GLFW_JOYSTICK_1, &gamepad))
812 return;
813 #define MAP_BUTTON(KEY_NO, BUTTON_NO, _UNUSED) do { io.AddKeyEvent(KEY_NO, gamepad.buttons[BUTTON_NO] != 0); } while (0)
814 #define MAP_ANALOG(KEY_NO, AXIS_NO, _UNUSED, V0, V1) do { float v = gamepad.axes[AXIS_NO]; v = (v - V0) / (V1 - V0); io.AddKeyAnalogEvent(KEY_NO, v > 0.10f, Saturate(v)); } while (0)
815#else
816 int axes_count = 0, buttons_count = 0;
817 const float* axes = glfwGetJoystickAxes(GLFW_JOYSTICK_1, &axes_count);
818 const unsigned char* buttons = glfwGetJoystickButtons(GLFW_JOYSTICK_1, &buttons_count);

853static void ImGui_ImplGlfw_UpdateMonitors()
854{
855 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
856 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
857 bd->WantUpdateMonitors = false;
858
859 int monitors_count = 0;
860 GLFWmonitor** glfw_monitors = glfwGetMonitors(&monitors_count);
861 if (monitors_count == 0) // Preserve existing monitor list if there are none. Happens on macOS sleeping (#5683)
862 return;
863
864 platform_io.Monitors.resize(0);
865 for (int n = 0; n < monitors_count; n++)
866 {
867 ImGuiPlatformMonitor monitor;
868 int x, y;
869 glfwGetMonitorPos(glfw_monitors[n], &x, &y);
870 const GLFWvidmode* vid_mode = glfwGetVideoMode(glfw_monitors[n]);
871 if (vid_mode == nullptr)
872 continue; // Failed to get Video mode (e.g. Emscripten does not support this function)
873 monitor.MainPos = monitor.WorkPos = ImVec2((float)x, (float)y);
874 monitor.MainSize = monitor.WorkSize = ImVec2((float)vid_mode->width, (float)vid_mode->height);
875#if GLFW_HAS_MONITOR_WORK_AREA
876 int w, h;
877 glfwGetMonitorWorkarea(glfw_monitors[n], &x, &y, &w, &h);
878 if (w > 0 && h > 0) // Workaround a small GLFW issue reporting zero on monitor changes: https://github.com/glfw/glfw/pull/1761
879 {
880 monitor.WorkPos = ImVec2((float)x, (float)y);
881 monitor.WorkSize = ImVec2((float)w, (float)h);
882 }
883#endif
884#if GLFW_HAS_PER_MONITOR_DPI
885 // Warning: the validity of monitor DPI information on Windows depends on the application DPI awareness settings, which generally needs to be set in the manifest or at runtime.
886 float x_scale, y_scale;
887 glfwGetMonitorContentScale(glfw_monitors[n], &x_scale, &y_scale);
888 monitor.DpiScale = x_scale;
889#endif
890 monitor.PlatformHandle = (void*)glfw_monitors[n]; // [...] GLFW doc states: "guaranteed to be valid only until the monitor configuration changes"
891 platform_io.Monitors.push_back(monitor);
892 }
893}

895void ImGui_ImplGlfw_NewFrame()
896{
897 ImGuiIO& io = ImGui::GetIO();
898 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
899 IM_ASSERT(bd != nullptr && "Did you call ImGui_ImplGlfw_InitForXXX()?");
900
901 // Setup display size (every frame to accommodate for window resizing)
902 int w, h;
903 int display_w, display_h;
904 glfwGetWindowSize(bd->Window, &w, &h);
905 glfwGetFramebufferSize(bd->Window, &display_w, &display_h);
906 io.DisplaySize = ImVec2((float)w, (float)h);
907 if (w > 0 && h > 0)
908 io.DisplayFramebufferScale = ImVec2((float)display_w / (float)w, (float)display_h / (float)h);
909 if (bd->WantUpdateMonitors)
910 ImGui_ImplGlfw_UpdateMonitors();
911
912 // Setup time step
913 // (Accept glfwGetTime() not returning a monotonically increasing value. Seems to happens on disconnecting peripherals and probably on VMs and Emscripten, see #6491, #6189, #6114, #3644)
914 double current_time = glfwGetTime();
915 if (current_time <= bd->Time)
916 current_time = bd->Time + 0.00001f;
917 io.DeltaTime = bd->Time > 0.0 ? (float)(current_time - bd->Time) : (float)(1.0f / 60.0f);
918 bd->Time = current_time;
919
920 ImGui_ImplGlfw_UpdateMouseData();
921 ImGui_ImplGlfw_UpdateMouseCursor();
922
923 // Update game controllers (if enabled and available)
924 ImGui_ImplGlfw_UpdateGamepads();
925}
