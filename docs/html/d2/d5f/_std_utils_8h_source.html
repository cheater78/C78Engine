65 {
66 std::string output = del;
67
68 uint32_t wordInd = 0;
69 for (std::string part : strs) {
70 if (wordInd < start) continue;
71 if (wordInd + 1 > count) break;
72 if (wordInd + 1 == strs.size() || wordInd + 2 > count)
